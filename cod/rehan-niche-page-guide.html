<!-- Full-page A/B · flag redirect · CANONICAL (control) pages only -->
<style>
  /* Hide the body until the flag resolves so users never see the control flash
     before being redirected to variant-b. Safety timeout reveals after 1500ms
     in case PostHog fails so nobody ever sits on a blank page. */
  html:not(.ph-ready) body { visibility: hidden; }
</style>
<script>
  // CHANGE THESE two per page. B URLs are confirmed; swap in per-page value from Step 10 table.
  var EXPERIMENT_KEY = 'niche-lp-educator-v1';
  var VARIANT_B_URL  = '/educator-page-page'; // B variant (confirmed)

  // Safety reveal so users are never stuck on a blank page
  setTimeout(function(){ document.documentElement.classList.add('ph-ready'); }, 1500);

  window.posthog = window.posthog || [];
  posthog.onFeatureFlags(function() {
    var variant = posthog.getFeatureFlag(EXPERIMENT_KEY) || 'control';

    // Fire exposure BEFORE redirecting so PostHog counts both arms
    posthog.capture('$feature_flag_called', {
      $feature_flag: EXPERIMENT_KEY,
      $feature_flag_response: variant,
      $variant_side: 'control-entry'
    });

    if (variant === 'variant-b') {
      // Preserve query string (utm_*, fbclid, gclid, hyros params) across the redirect
      var qs = location.search || '';
      location.replace(VARIANT_B_URL + qs);
      return;
    }

    // User stays on control — reveal the page
    document.documentElement.classList.add('ph-ready');
  });
</script>
<!-- /Flag redirect -->
Why location.replace not location.href: replace does not push a history entry, so the back button skips the canonical URL and goes straight to where the user came from. Avoids a weird browser-history loop.
Why exposure fires before redirect: if we redirected first, we'd miss the exposure event on the variant-b arm because the page unloads too fast. Firing first guarantees PostHog counts every user on both arms.
